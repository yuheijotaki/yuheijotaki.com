function change_posts_per_page($query) {
	/* 管理画面,メインクエリに干渉しないために必須 */
	if( is_admin() || ! $query->is_main_query() ){
		return;
	}
	/* ここからカスタム */
	if ( $query->is_tax('TAXONOMY_NAME') ) {
		$query->set( 'posts_per_page', '-1');
		return;
	}
}
add_action( 'pre_get_posts', 'change_posts_per_page' );
