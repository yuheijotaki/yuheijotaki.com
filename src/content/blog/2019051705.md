<h3>フォーム要素をバインドしてスタイル変更</h3>

<details><summary><strong><code>App.vue</code></strong>（一部簡略化）</summary><div>

```html
<template>
  <div id="app">
    <div>
      <inputFontSize @change="changeFontSize"></inputFontSize>
    </div>
    <div>
      <div v-bind:style="styleObject">ここにテキストが入ります。</div>
    </div>
  </div>
</template>

<script>
import inputFontSize from './components/input-fontSize'

export default {
  name: 'App',
  components: {
    inputFontSize
  },
  data() {
    return {
      styleObject: {
        color: 'red',
        fontSize: '16px'
      }
    }
  },
  methods: {
    changeFontSize: function (size) {
      this.styleObject.fontSize = `${size}px`;
    }
  }
}
</script>
```

</div></details>

<details><summary><strong><code>input-fontSize.vue</code></strong>（一部簡略化）</summary><div>

```html
<template>
  <div>
    FontSize: <input
      type="number"
      min="10"
      :value="fontSize"
      @input="updateValue"
      @focus="$emit('focus', $event)"
      @blur="$emit('blur', $event)"
    > px
  </div>
</template>

<script>
export default {
  name: 'inputFontSize',
  data() {
    return {
      fontSize: 16,
    }
  },
  methods: {
    updateValue(e) {
      this.$emit('input', e.target.value)
      this.$emit('change', e.target.value)
    }
  }
}
</script>
```

</div></details>

<h3>style オブジェクト を使ってインラインスタイル</h3>

<a href="https://jp.vuejs.org/v2/guide/class-and-style.html#%E3%82%AA%E3%83%96%E3%82%B8%E3%82%A7%E3%82%AF%E3%83%88%E6%A7%8B%E6%96%87-1">公式ドキュメント</a> の通りですが、

```js
data() {
  return {
    styleObject: {
      color: 'red',
      fontSize: '16px'
    }
  }
}
```

と定義して

```html
<div v-bind:style="styleObject"></div>
```

のようにバインドできる。  
親側の変更監視のメソッドは

```js
methods: {
  changeFontSize: function (size) {
    this.styleObject.fontSize = `${size}px`;
  }
}
```

とする。
