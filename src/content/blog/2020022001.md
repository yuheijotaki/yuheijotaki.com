<h2>Pugで <code>v-if</code></h2>

Pugで <code>v-if</code> を使う

```js
<template lang="pug">
  div
    p(v-if="loading") loading
    p(v-else) loaded
</template>

<script>
import axios from "axios";
export default {
  data () {
    return {
      posts: [],
      loading: true
    }
  },
  mounted :function(){
    axios.get( 'https://works.yuheijotaki.com/wp-json/wp/v2/posts?per_page=100' )
    .then( response => {
      this.posts = response.data
      this.loading = false
    })
    .catch( error => {
      console.log(error)
    })
  }
}
</script>
```
